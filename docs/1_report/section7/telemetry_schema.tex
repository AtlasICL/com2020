\subsection{Definition of JSON Schema}
This section defines the structure of the telemetry events generated by the Java application and parsed by the Python application. All events are stored in JSON format and must adhere to the following key-value constraints.

\subsubsection{Valid Value Enumerations}
The following values are the only accepted strings for their respective categories within the telemetry system.

\paragraph{List of Valid Event Types}
\begin{itemize}
    \item \texttt{StartSession}
    \item \texttt{NormalEncounterStart}
    \item \texttt{NormalEncounterComplete}
    \item \texttt{NormalEncounterFail}
    \item \texttt{BossEncounterStart}
    \item \texttt{BossEncounterComplete}
    \item \texttt{BossEncounterFail}
    \item \texttt{GainCoin}
    \item \texttt{BuyUpgrade}
    \item \texttt{EndSession}
    \item \texttt{SettingsChange}
    \item \texttt{KillEnemy}
\end{itemize}

\paragraph{List of Valid Difficulties}
\begin{itemize}
    \item \texttt{Easy}
    \item \texttt{Normal}
    \item \texttt{Hard}
\end{itemize}

\paragraph{List of Valid Encounter Names}
\begin{itemize}
    \item \texttt{GoblinEncounter}
    \item \texttt{FishmanEncounter}
\end{itemize}

\paragraph{List of Valid Upgrade Names}
\begin{itemize}
    \item \texttt{PhysicalDamageResistance}
    \item \texttt{FireDamageResistance}
    \item \texttt{WaterDamageResistance}
    \item \texttt{ThunderDamageResistance}
    \item \texttt{ImprovedPhysicalDamage}
    \item \texttt{ImprovedFireDamage}
    \item \texttt{ImprovedWaterDamage}
    \item \texttt{ImprovedThunderDamage}
    \item \texttt{SlashUnlock}
    \item \texttt{AbsolutePulseUnlock}
    \item \texttt{WaterJetUnlock}
    \item \texttt{FireBallUnlock}
    \item \texttt{ThunderStormUnlock}
\end{itemize}

\paragraph{List of Valid Enemy Names}
\begin{itemize}
    \item \texttt{Goblin}
    \item \texttt{Fishman}
\end{itemize}

\paragraph{List of Valid Settings}
\begin{itemize}
    \item \texttt{MaxStageReached}
    \item \texttt{PlayerMaxHealth}
    \item \texttt{EnemyDamageMultiplier}
    \item \texttt{StartingLives}
    \item \texttt{MaxMagic}
    \item \texttt{MagicRegenRate}
    \item \texttt{ShopItemCount}
\end{itemize}
