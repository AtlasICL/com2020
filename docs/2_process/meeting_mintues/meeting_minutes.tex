    \textbf{Date:} 14/01/2026 \\

    \textbf{Attendees:} Luca Pacitti, Luca Croci, Emre Acarsoy, Harry Taylor, Tom Croft, Will Finney, Kazybek Khairulla

    \textbf{Agenda Items}
    \begin{itemize}
        \item Discussion of gameplay design.
        \item Discussion of three mechanics to influence success.
        \item Strategy for agile approach to the project.
    \end{itemize}

    \textbf{Gameplay Design}
    \begin{itemize}
        \item 2D turn-based RPG.
        \item Simple to understand.
        \item Three difficulty levels, with different default parameters for each.
        \item 10 levels total:
        \begin{itemize}
            \item Boss levels at 3, 6, 9, and 10.
            \item Difficulty increases over time.
        \end{itemize}
        \item Defeating enemies grants coins; defeating bosses grants more coins.
        \item Every level is an encounter.
        \item After each encounter, the player enters a shop menu where coins can be spent on upgrades.
        \item The player has a number of lives, which varies depending on the selected difficulty.
        \begin{itemize}
            \item If the player dies and has lives remaining, they restart the current level.
            \item If the player dies with no lives remaining, they restart the game entirely.
        \end{itemize}
        \item Each encounter starts with the player at maximum health.
        \item Automated agent will be used to play the game as a simulation, to demonstrate effects of tweaking design parameters.
    \end{itemize}

    \textbf{Mechanics to Influence Success}
    \begin{itemize}
        \item \textbf{Coins:} used to purchase upgrades after an encounter.
        \item \textbf{Magic:} used to execute actions during an encounter.
        \item \textbf{Health:} used to evaluate whether a player is alive during an encounter.
    \end{itemize}

    \textbf{Agile Approach to Project}
    \begin{itemize}
        \item Two sprints:
        \begin{itemize}
            \item \textbf{First sprint:} from now until the CW1 deadline.
            \item \textbf{Second sprint:} from the CW1 deadline until the CW2 deadline.
        \end{itemize}

        \item Requirements to be composed using the \textbf{MoSCoW} prioritisation structure:
        \begin{itemize}
            \item First sprint focuses on \textbf{Must-have} goals for the prototype.
            \item Second sprint focuses on \textbf{Should-have} and \textbf{Could-have} goals.
        \end{itemize}
    \end{itemize}
