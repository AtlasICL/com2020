\subsection{From Source}
\subsubsection{Common}
These steps are required for both applications.

\begin{enumerate}
  \item Install python 3.13 or later. Instructions can be found \href{https://wiki.python.org/moin/BeginnersGuide/Download}{here}.
    \item Install requests: \verb|pip install requests|
  \item Setup environment variables. First obtain them from the email that is sent to all clients, then follow the below steps:
    \begin{itemize}
      \item On Windows run:
        \begin{enumerate}
            \item \verb|$env:OIDC_ISSUER="<ENVIRONMENT VARIABLE VALUE>"|
            \item \verb|$env:OIDC_CLIENT_ID="<ENVIRONMENT VARIABLE VALUE>"|
            \item \verb|$env:OIDC_CLIENT_SECRET="<ENVIRONMENT VARIABLE VALUE>"|
        \end{enumerate}
      \item On macOS or Linux run:
        \begin{enumerate}
            \item \verb|export OIDC_ISSUER="<ENVIRONMENT VARIABLE VALUE>"|
            \item \verb|export OIDC_CLIENT_ID="<ENVIRONMENT VARIABLE VALUE>"|
            \item \verb|export OIDC_CLIENT_SECRET="<ENVIRONMENT VARIABLE VALUE>"|
        \end{enumerate}
    \end{itemize}
\end{enumerate}

\subsubsection{Python Application}
\label{sec:python_src_deployment}
\begin{enumerate}
    \item Enter the telemetry directory: \verb|cd telemetry|
    \item Create a virtual environment: \verb|python3 -m venv venv|
  \item Activate the virtual environment:
    \begin{itemize}
        \item On windows: \verb|./venv/Scripts/Activate|
        \item On macOS or Linux: \verb|source venv/bin/activate|
    \end{itemize}
    \item Install packages: \verb|pip install -r requirements.txt|
  \item Run the program:
    \begin{itemize}
        \item On windows: \verb|python telemetry_app.py|
        \item On macOS or Linux: \verb|python3 telemetry_app.py|
    \end{itemize}
    \item To exit the venv run: \verb|deactivate|
\end{enumerate}

\subsubsection{Java Application}
\label{sec:java_src_deployment}
\begin{enumerate}
  \item Install java 17+. Instructions can be found \href{https://www.java.com/en/download/manual.jsp}{here}
  \item Install maven. Instructions can be found \href{https://maven.apache.org/install.html}{here}
    \item Enter the game directory \verb|cd game|
    \item Run \verb|mvn clean compile exec:java "-Dexec.mainClass=wizardquest.ui.GameRunPage"|
\end{enumerate}
